# HSE06 hybrid functional INCAR template
# IMPORTANT: Pre-converge with PBE first and keep WAVECAR to speed up HSE06
# HSE06 is 20-100x more expensive than PBE

# Hybrid functional settings

# AEXX   = 0.25      # default 25% exact exchange; tune for specific materials
ALGO     = Damped    # most robust for hybrids; or All (faster if it converges)
TIME     = 0.4       # step size for Damped algorithm
PRECFOCK = Fast      # FFT precision for Fock exchange; Fast is usually sufficient
ISYM     = 3         # symmetrize charge density (recommended for hybrids)
NELM     = 200       # hybrids converge slowly

ISMEAR = 0
SIGMA  = 0.05

LORBIT = 11
